        <li>First name,</li>
        <li>Surname,</li>
        <li>Email,</li>
        <li>Password</li>
    </ul>
  </li>
  <li>Login/Logout
    <ul>

      <li>Successful login takes to welcome dashboard where options are:</li>
      <ul>
        <li>View timetables</li>

    </ul>

  <li>Edit user details, add photo</li>
  <li>Floor Map
    <ul>

    </ul>
  </li>
  <li>Timetable</li>

      <ul>
        <li>Business</li>
        <li>Computing </li>

        <li>Economics</li>
